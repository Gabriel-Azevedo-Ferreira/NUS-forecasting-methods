Regression on Time¶

A trend model:¶

No other covariates except time in the regression model

  1. no trend
  2. linear trend
  3. quadratic trend

Detecting Autocorrelation:¶

Durbin-Watson Test

Seasonal Variation¶

  1. constant (addictive) seasonal variation: dummy variables or trigonometric functions
  2. multiplicative seasonal variation : transformation

def transform_exog_to_model(fit, exog):
    transform=True
    self=fit

    # The following is lifted straight from statsmodels.base.model.Results.predict()
    if transform and hasattr(self.model, 'formula') and exog is not None:
        from patsy import dmatrix
        exog = dmatrix(self.model.data.orig_exog.design_info,
                       exog)

    if exog is not None:
        exog = np.asarray(exog)
        if exog.ndim == 1 and (self.model.exog.ndim == 1 or
                               self.model.exog.shape[1] == 1):
            exog = exog[:, None]
        exog = np.atleast_2d(exog)  # needed in count model shape[1]

    # end lifted code
    return exog

import math
def trigonometric_transform(data, colName, period, order):
    val = data[colName].values;
    pi = math.pi;
    
    # loop for each order
    for i in range(1, order+1):
        n1 = 'Sin{}'.format(i);
        n2 = 'Cos{}'.format(i);
        data[n1] = np.sin(i*2*pi*val/period);
        data[n2] = np.cos(i*2*pi*val/period);

    return data;
